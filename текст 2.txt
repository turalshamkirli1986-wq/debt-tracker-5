    function toAZN(amount, fromCurr) { if (fromCurr === 'AZN') return amount; return amount * exchangeRates[fromCurr]; }
    function fromAZN(amountAZN, toCurr) { if (toCurr === 'AZN') return amountAZN; return amountAZN / exchangeRates[toCurr]; }
    function formatMoney(amountAZN, curr) { let val = fromAZN(amountAZN, curr); let sym = { AZN:'₼', USD:'$', EUR:'€', TRY:'₺', RUB:'₽' }; return `${val.toLocaleString(undefined, {minimumFractionDigits:2, maximumFractionDigits:2})} ${sym[curr] || curr}`; }

    function loadFromLocal() {
        let stored = localStorage.getItem(STORAGE_KEY);

    // Azalt/Artır məntiqi (isIncrease = true -> məbləği artır, false -> azaldır)
    function updateDebtAmount(debtId, amountAZN, isIncrease) {
        let idx = debts.findIndex(d => d.id === debtId);
        if (idx === -1) return false;
        let current = debts[idx].amount;
        let sign = Math.sign(current);
        if (sign === 0) sign = 1;
        let newAmount;
        if (isIncrease) newAmount = current + sign * amountAZN;
        else newAmount = current - sign * amountAZN;
        if (Math.abs(newAmount) < 0.01) newAmount = 0;
        if (newAmount === 0) {
            if (confirm('Bu borcu silmək istəyirsiniz?')) { debts.splice(idx, 1); updateAll(); closeEditModal(); return true; }
            else return false;
        }
        debts[idx].amount = newAmount;
        if (!debts[idx].transactions) debts[idx].transactions = [];
        let comment = document.getElementById('transactionComment').value || (isIncrease ? 'Artırma' : 'Azaltma');
        debts[idx].transactions.push({ id: Date.now(), dateStr: getDateTime(), delta: newAmount - current, newBalance: newAmount, comment: comment });
        updateAll();
        return true;
    }

    function addNewDebt(name, amount, currency, type) {
        if (!name.trim()) { alert('Ad daxil edin'); return false; }
        let num = parseFloat(amount);
        if (isNaN(num) || num === 0) { alert('Məbləğ daxil edin'); return false; }
        let azn = toAZN(Math.abs(num), currency);
        let final = type === 'receivable' ? azn : -azn;
        let newDebt = { id: Date.now(), name: name.trim(), amount: final, currency: currency, transactions: [] };
        newDebt.transactions.push({ id: Date.now(), dateStr: getDateTime(), delta: final, newBalance: final, comment: 'İlkin' });
        debts.push(newDebt);
        updateAll();
        return true;
    }
